\begin{abstract}
We present a framework for stochastic projected dynamical systems whose constraints are driven by real‑world data and whose control laws adapt to both stress and uncertainty. The state is steered toward a data‑defined feasible set, while a ``mercy'' mechanism temporarily relaxes these constraints under high stress, tempered by a confidence signal that restores strictness when appropriate. A resistance (brake) term, inversely related to uncertainty, prevents reckless movement in volatile conditions. The projection is softened through a proximal penalty that maintains firm non‑expansiveness, and an external feedback loop from the environment corrects drift. We prove that under mild conditions the system remains mean‑square stable around the moving truth, with an explicit bound that separates noise, data shift, and relaxation errors. The design anchors classical Lyapunov analysis to measurable, real‑time quantities, yielding an adaptive, self‑correcting intelligence kernel.
\end{abstract}

\begin{proposition}[Firm non‑expansiveness]
\label{prop:firm}
For any closed convex set $\mathcal{C}$ and any $x,y$,
\[
\|\Pi_{\mathcal{C}}(x) - \Pi_{\mathcal{C}}(y)\|^2 \le \|x - y\|^2
- \|(x - \Pi_{\mathcal{C}}(x)) - (y - \Pi_{\mathcal{C}}(y))\|^2.
\]
In particular, for $y \in \mathcal{C}$,
\[
\|\Pi_{\mathcal{C}}(x) - y\|^2 \le \|x - y\|^2 - \|x - \Pi_{\mathcal{C}}(x)\|^2.
\]
\end{proposition}

\begin{lemma}[Relaxation error]\label{lem:relax}
Let $\mathcal{M}$ be a closed convex set and $\mathcal{M}^{\varepsilon} = \{x \mid \operatorname{dist}(x,\mathcal{M}) \le \varepsilon\}$ be its uniform $\varepsilon$‑relaxation. Then for any $F$,
\[
\|\Pi_{\mathcal{M}^{\varepsilon}}(F) - \Pi_{\mathcal{M}}(F)\| \le \varepsilon .
\]
\end{lemma}
\begin{proof}
Since $\mathcal{M}^{\varepsilon} = \mathcal{M} + \varepsilon B$, where $B$ is the closed unit ball, the projection onto $\mathcal{M}^{\varepsilon}$ can be expressed as
\[
\Pi_{\mathcal{M}^{\varepsilon}}(F) = \Pi_{\mathcal{M}}(F) + \delta(F),
\]
where $\delta(F)$ is the vector of minimal length that moves $\Pi_{\mathcal{M}}(F)$ into $\mathcal{M}^{\varepsilon}$ while satisfying the projection optimality. By the definition of the relaxation, $\|\delta(F)\| \le \varepsilon$, and the equality holds as a consequence of the Moreau decomposition. Hence the bound follows.
\end{proof}

\section{One‑Step Drift Analysis}
\label{sec:drift}

\begin{lemma}[One‑step drift with explicit constants]\label{lem:drift}
Assume $0<\alpha<1$, $0\le\beta_t\le\beta_0$, and the uncertainty $U_t$ is bounded. Define
\[
\Delta_t := \operatorname{dist}_H(\mathcal{M}_t,\mathcal{M}_{t+1}) .
\]
Then the conditional expectation of the Lyapunov function satisfies
\[
\mathbb{E}[V_{t+1}\mid\mathcal{F}_t]
\le \bigl(1 - \alpha\beta_t + \alpha^2(1+\beta_0)^2\bigr) V_t
   + \alpha(1+\beta_0)(\varepsilon_t+\Delta_t)\bigl(4\sqrt{V_t} + 2(\varepsilon_t+\Delta_t)\bigr)
   + \sigma^2 + \gamma^2 G_{\max}^2 .
\]
\end{lemma}
\begin{proof}
Rewrite the dynamics as
\[
F_{t+1} - x_{t+1} = (1-\alpha\beta_t)(F_t - x_t) + \alpha(1+\beta_t)(\Pi_t - x_t) + (x_t - x_{t+1}) + \xi_t + \gamma G_t .
\]
Squaring and taking conditional expectation eliminates the martingale noise. Using firm non‑expansiveness for the inner product $\langle F_t - x_t, \Pi_t - F_t\rangle$ and Lemma~\ref{lem:relax} to replace $\|\Pi_t - x_t\|$ by $\varepsilon_t$, we obtain after straightforward algebraic manipulation
\[
\mathbb{E}[V_{t+1}\mid\mathcal{F}_t]
\le V_t - \alpha\beta_t V_t + \alpha^2(1+\beta_0)^2 V_t
   + \alpha(1+\beta_0)(\varepsilon_t+\Delta_t) \bigl(4\sqrt{V_t} + 2(\varepsilon_t+\Delta_t)\bigr)
   + \sigma^2 + \gamma^2 G_{\max}^2 .
\]
The constants $A=(1+\beta_0)^2$, $B=4(1+\beta_0)$, $C=2(1+\beta_0)$ appear explicitly when bounding the cross terms.
\end{proof}

\section{Main Stability Theorem}
\label{sec:stability}

\begin{assumption}[Bounded uncertainty away from zero]\label{ass:Umin}
The uncertainty process $\{U_t\}$ satisfies $0 < U_{\min} \le U_t \le U_{\max}$ for all $t$.
\end{assumption}

\begin{theorem}[Reality‑Grounded Mean‑Square Stability]\label{thm:stability}
Suppose $\alpha(1+\beta_0)<1$, the relaxation $\varepsilon_t$ and the constraint drift $\Delta_t$ satisfy $\varepsilon_t\to0$, $\Delta_t\to0$ with $\sum_t (\varepsilon_t^2+\Delta_t^2)<\infty$, and Assumption~\ref{ass:Umin} holds. Then there exist $\rho\in(0,1)$ and $D\ge0$ such that
\[
\limsup_{t\to\infty} \mathbb{E}[V_t] \le \frac{D}{1-\rho}.
\]
\end{theorem}
\begin{proof}
From Lemma~\ref{lem:drift} and the bound $\beta_t \ge \beta_{\min} = \beta_0\frac{U_{\min}}{U_0+U_{\min}}>0$, apply Young's inequality $2ab \le \frac{1}{\theta}a^2 + \theta b^2$ with $\theta = \frac12 \alpha\beta_{\min}$ to the $\sqrt{V_t}$ term. After grouping, we obtain for sufficiently large $t$,
\[
\mathbb{E}[V_{t+1}\mid\mathcal{F}_t] \le \rho V_t + D_t,
\]
with $\rho = 1 - \frac12 \alpha\beta_{\min} + \alpha^2(1+\beta_0)^2 \in (0,1)$ (by taking $\alpha$ small enough) and
$D_t = \bigl(\frac{4(1+\beta_0)^2}{\beta_{\min}} + 2(1+\beta_0)\bigr) \alpha(\varepsilon_t+\Delta_t)^2 + \sigma^2 + \gamma^2 G_{\max}^2$.
The sum of $D_t$ over $t$ is bounded because $\sum (\varepsilon_t+\Delta_t)^2<\infty$; hence the tail average is finite. Taking unconditional expectation and iterating gives the $\limsup$ bound.
\end{proof}

\section{Related Work}
\label{sec:related}

Projected dynamical systems were formalized by Bena\"im~\cite{benaim1999} in the context of stochastic approximation, and the firm non‑expansiveness of projection operators is a cornerstone of convex analysis~\cite{bauschke2011convex}. The combination of time‑varying constraints with adaptive relaxation draws on the relax‑and‑project philosophy of Rockafellar and Wets~\cite{rockafellar2009}. Recent work on online convex optimization with moving constraints~\cite{kushner2003} provides convergence results in expectation, but typically does not include the uncertainty‑modulated resilience mechanisms introduced here.

\section{Conclusion}
\label{sec:conclusion}

We have grounded the abstract projected stochastic system in measurable, real‑time signals—stress, uncertainty, confidence—and proved its mean‑square stability with explicit constants. The resulting dynamics possess the ability to tighten constraints when confident, relax them under duress, and brake in volatile conditions, while staying anchored to external ground truth. This architecture is a rigorous mathematical kernel for self‑correcting AI.
