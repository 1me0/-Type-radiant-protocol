\begin{abstract}
Any adaptive system can be characterized by a regulatory gain \(\alpha\in(0,1]\) and a disturbance amplification factor \(\beta\ge0\).
The product \(\Lambda(t)\equiv\alpha(t)(1+\beta(t))\) defines the system's instantaneous load.
We prove that the static condition \(\Lambda<1\) is necessary and sufficient for mean‑square stability under the small‑gain theorem,
and that its preservation over time requires the logarithmic rate of regulation to dominate the combined growth of disturbance and self‑amplification:
\[
\frac{\dot\alpha}{\alpha} + \frac{\dot\beta}{1+\beta} < 0 .
\]
From this single inequality we derive a complete phase diagram (stability–criticality–collapse) and a family of universal laws that unify
Ashby's requisite variety, the thermodynamics of open systems, Fisher's fundamental theorem, and economic stability.
A toy model illustrates the principle quantitatively.
\end{abstract}

% =================================================================
\section{The Static Condition}
% =================================================================
Consider a feedback loop where a controller of gain \(\alpha\) counteracts a disturbance that is amplified by a factor \(1+\beta\) over one characteristic time.
The small‑gain theorem~\cite{zames1966} guarantees that the closed‑loop system is input‑output stable if the loop gain satisfies
\begin{equation}\label{eq:static}
\boxed{\;\alpha(1+\beta) < 1\;}.
\end{equation}
This inequality delimits a safe operating region in the \((\alpha,\beta)\) plane.

\begin{theorem}[Static stability]\label{thm:static}
For a plant with finite \(L_2\)‑gain \(\le 1+\beta\) and a linear controller of gain \(\alpha\), stability is equivalent to \(\alpha(1+\beta)<1\).
\end{theorem}

\section{The Dynamic Extension}
% =================================================================
In realistic systems \(\alpha\) and \(\beta\) evolve over time.
Define the instantaneous load \(\Lambda(t)=\alpha(t)(1+\beta(t))\).
To remain stable, the system must not cross the boundary \(\Lambda=1\).
A sufficient condition for this is that the load does not increase whenever it approaches the critical threshold:
\[
\dot\Lambda(t) = \dot\alpha(t)(1+\beta(t)) + \alpha(t)\dot\beta(t) < 0 \qquad\text{when }\Lambda(t)\ge 1-\varepsilon .
\]
Dividing by the positive quantity \(\alpha(1+\beta)\) yields the dimensionless condition
\begin{equation}\label{eq:dynamic}
\boxed{\;\frac{\dot\alpha}{\alpha} + \frac{\dot\beta}{1+\beta} < 0\;}.
\end{equation}
Equivalently,
\[
\frac{d}{dt}\ln\Lambda(t) < 0 .
\]

\begin{theorem}[Dynamic Stability Preservation]\label{thm:dynamic}
Assume \(\alpha(t)>0\), \(\beta(t)\ge0\) are \(C^1\) and \(\Lambda(0)<1\).
If \(\frac{\dot\alpha}{\alpha} + \frac{\dot\beta}{1+\beta} < 0\) holds for all \(t\ge0\), then \(\Lambda(t)<1\) for all \(t\ge0\), and the system remains stable.
\end{theorem}

\section{Logarithmic Formulation and Phase Diagram}
% =================================================================
The inequality~\eqref{eq:dynamic} can be written symmetrically as
\begin{equation}\label{eq:log}
\boxed{\;\partial_t\ln\alpha < -\,\partial_t\ln(1+\beta)\;},
\qquad \alpha>0,\;\beta\ge 0 .
\end{equation}
This formulation reveals three distinct regimes:
\begin{enumerate}
\item \textbf{Stable regime}: \(\partial_t\ln\alpha < -\partial_t\ln(1+\beta)\).  
      The load \(\Lambda\) decays or stays safely below \(1\).
\item \textbf{Critical boundary}: \(\partial_t\ln\alpha = -\partial_t\ln(1+\beta)\).  
      Regulation and disturbance evolve such that \(\Lambda\) is stationary — the system hovers exactly on the stability margin.
\item \textbf{Collapse regime}: \(\partial_t\ln\alpha > -\partial_t\ln(1+\beta)\).  
      The load grows monotonically; once \(\Lambda\ge 1\), contraction is lost and the system collapses.
\end{enumerate}

\noindent\textbf{Meta‑Axiom.} The single overarching law that follows from the above is:
\begin{equation}\label{eq:meta}
\boxed{\;\text{A system survives only when its regulatory decay outpaces the combined effect of disturbance growth and self‑amplification.}\;}
\end{equation}

\section{The Meta‑Theorem of Regulatory Dominance over Entropy}
% =================================================================
The results of the previous sections can be compactly summarised in a single theorem.

\begin{theorem}[Law of Regulatory Dominance]\label{thm:meta}
Let \(\alpha(t)\in(0,1]\) and \(\beta(t)\ge0\) be differentiable functions, and define \(\Lambda(t)=\alpha(t)(1+\beta(t))\).
\begin{enumerate}
  \item[(i)] \textbf{Static Stability.} If \(\Lambda(t)<1\) then the effective amplification gain \(G=\alpha(t)(1+\beta(t))\) satisfies \(G<1\).  Consequently, any Lyapunov function of the system contracts exponentially, and the system is asymptotically stable in mean square.
  \item[(ii)] \textbf{Dynamic Persistence.} Assume \(\Lambda(t_0)<1\) for some initial time \(t_0\).  Then a sufficient condition for \(\Lambda(t)<1\) to hold for all \(t>t_0\) is
  \[
  \frac{d}{dt}\ln\Lambda(t) < 0
  \qquad\Longleftrightarrow\qquad
  \frac{\dot\alpha}{\alpha} + \frac{\dot\beta}{1+\beta} < 0 .
  \]
  That is, the relative decay of regulation must be steeper than the relative growth of the amplified disturbance.
  \item[(iii)] \textbf{Collapse Criterion.} If \(\frac{d}{dt}\ln\Lambda(t)\ge 0\) over a time interval longer than the system's characteristic response time \(\tau\), then \(\Lambda(t)\) is non‑decreasing and will cross the critical threshold \(\Lambda=1\) in finite time.  Once \(\Lambda\ge 1\), the contraction property is lost, and the system enters an irreversible cascade toward collapse.
\end{enumerate}
\end{theorem}

\begin{proof}[Sketch of proof]
Part (i) follows from the Subcritical Gain Contraction Theorem: for the discrete update
\[
x_{t+1} = x_t + \alpha\nabla\Phi\big((1+\beta)\widetilde{\Pi}(x_t)-\beta x_t\big) ,
\]
the error norm contracts by a factor at most \(G\) if \(G<1\).  The continuous analogue gives the same condition.
Part (ii) is obtained by logarithmic differentiation of \(\Lambda<1\), and the inequality \(\dot\alpha/\alpha+\dot\beta/(1+\beta)<0\) ensures \(\ln\Lambda\) decreases monotonically.
Part (iii) is a consequence of the mean‑value theorem applied to \(\ln\Lambda(t)\): if the derivative is non‑negative for a duration exceeding \(\tau\), the function must eventually reach \(\ln\Lambda=0\) (i.e., \(\Lambda=1\)), where the contraction guarantee vanishes.
\end{proof}

\begin{remark}
This theorem unifies Ashby's Law of Requisite Variety (control must match external variety), the Second Law of Thermodynamics for open systems (entropy export must exceed internal generation), Fisher's Fundamental Theorem of Natural Selection (the rate of fitness increase equals the additive genetic variance), and the stability margins of control engineering.  It is the mathematical boundary between adaptive persistence (sustained negative entropy production) and entropic dissolution.
\end{remark}

\section{From the Inequality to Universal Laws}
% =================================================================
The static and dynamic inequalities encapsulate a set of survival laws that are observed across disciplines.

\begin{corollary}[Regulation must decay faster than disturbance amplifies]
\label{cor:growth}
Stability requires \(\dot\alpha/\alpha + \dot\beta/(1+\beta) < 0\); i.e., the relative decay of regulation must exceed the relative growth of the amplified disturbance.
\end{corollary}

\begin{corollary}[Stability is a maintained inequality]
\label{cor:notstate}
A system stable today (\(\Lambda<1\)) can become unstable tomorrow if the dynamic inequality is violated.
Thus \emph{stability is not a state but a permanently maintained inequality}.
\end{corollary}

\begin{corollary}[Unregulated growth is dangerous]
\label{cor:unreg}
If \(\beta\) grows while \(\alpha\) remains fixed, the load \(\Lambda\) increases and the system destabilizes.
Hence \emph{growth itself is not dangerous — unregulated growth is}.
\end{corollary}

\begin{corollary}[Critical threshold sensitivity]
\label{cor:threshold}
Near \(\Lambda=1\), a tiny increase in \(\beta\) can push \(\Lambda\) above \(1\), causing disproportionate collapse.
\end{corollary}

\begin{corollary}[Resilience absorbs disturbance without losing identity]
\label{cor:resilience}
\(\alpha\) quantifies the capacity to absorb perturbation while preserving function.
For given \(\alpha\), the maximum tolerable \(\beta\) is \((1-\alpha)/\alpha\) in the static case.
\end{corollary}

\section{Universality Across Domains}
% =================================================================
The same mathematics governs every field where a balancing force confronts an amplifying challenge.

\begin{itemize}
\item \textbf{Control theory}: \(\alpha\) is the feedback gain, \(1+\beta\) the plant gain; stability is the small‑gain condition.
\item \textbf{Ecology}: \(\alpha\) represents density‑dependent mortality, \(\beta\) the intrinsic growth rate; coexistence requires \(\alpha(1+\beta)<1\), i.e., the community matrix's spectral radius \(<1\).
\item \textbf{Economics}: \(\alpha\) is the stabilisation policy multiplier, \(\beta\) the velocity of capital flight; the Samuelson multiplier‑accelerator model is stable only when \(\alpha(1+\beta)<1\) \cite{samuelson1939}.
\item \textbf{Cognitive regulation}: prefrontal inhibitory control (\(\alpha\)) must outweigh limbic reactivity (\(\beta\)); resilience deteriorates when regulation decays too slowly.
\item \textbf{Institutional viability}: governing capacity (\(\alpha\)) versus institutional decay/overload (\(\beta\)) obey the same law; collapse is predicted when \(\Lambda\) approaches \(1\) without corrective adaptation.
\end{itemize}

\section{Conclusion}
The Meta‑Theorem of Regulatory Dominance over Entropy provides a single mathematical criterion for the survival of any adaptive system.
The static condition \(\alpha(1+\beta)<1\) and the dynamic law \(\dot\alpha/\alpha + \dot\beta/(1+\beta) < 0\) define the boundary between ordered persistence and entropic collapse.
The three regimes—stability, criticality, and collapse—are a complete phase diagram of resilience, unifying control, ecology, economics, and cognition.
